028     * Starts the game and keeps it running until a winner is found or the gameboard is full. Later announces the winner.
029     * @param args the command line arguments
030     */
031    public static void main(String [] args) {
032

034        // NOTE: The code is filled with line breaks (System.out.println(); with nothing to print), for the sake of appeal and readability of the output.
035
036
037        MarkAmount(gameboardWidth, gameboardHeight);
038        PrintNewGameboard();

040        System.out.println();
041        System.out.println("Let's start the game!");
042        System.out.println();
043        // A delay (of 1000 milliseconds = 1 second) to help with the pacing and the readability of the output. Also found later in code with different values.
044        Wait(1000);
045        // Loops the game via PlayGame() until a win or a tie isn't found.
046        // Upon finding a victory/tie, ends the game and the end result is printed out.
047        do {
048            // Contains the player and computer turns + victory checking.
049            PlayGame();
050            // CheckWins() sets a char value for the variable "win".

052            // When the "win" variable is set as the mark of one of the players, the game ends.
053            if (win != markEmpty) {
054                hasWon = true;
055                // If "win" receives a mark of the human player, the game ends with an announcement of player victory.
056                if (win == markPlayer) {
057                    System.out.println();
058                    System.out.println("+--------------------------------------------------+");
059                    System.out.println("|           The player has won the game!           |");
060                    System.out.println("+--------------------------------------------------+");
061                // If "win" receives a mark of the computer player, the game ends with an announcement of computer victory.
062                } else if (win == markCPU) {
063                    System.out.println();
064                    System.out.println("+--------------------------------------------------+");
065                    System.out.println("|     The computer has (somehow) won the game!     |");
066                    System.out.println("+--------------------------------------------------+");
067                }
068            }
069            // If "isDraw" is true and no wins have been found, the game ends with an announcement of a tie.
070            if (isDraw && !hasWon) {
071                System.out.println();
072                System.out.println("+--------------------------------------------------+");
073                System.out.println("|              The game ends in a tie!             |");
074                System.out.println("+--------------------------------------------------+");
075            }
076        } while (!isDraw && !hasWon);
077    }

161        } else {
162            shorter = gameboardHeight;
163        }        
164        System.out.println("Now input the amount of marks required to win:");
165        // If user has set the size for gameboard as 10x10 or larger, the output suggests the value for the marks required to win to be at least 5.
166        // Minimum value is 3, as the minimum width/height is also 3.
167        if (gameboardWidth >= 10 && gameboardHeight >= 10) {
168            System.out.println("(Minimum: 3 / Maximum: " + shorter + " / Recommended: 5+)");
169        } else {
170        System.out.println("(Minimum: 3 / Maximum: " + shorter + ")");
171        }
172        // The value for the marks required to win is sent as parameters to, and received from, a method in the Validate class.

364    /**
365     * Checks if the marks placed by the user/computer form a horizontal, vertical or a diagonal line,
366     * containing an equal amount of that player's marks to the user defined amount of marks required to win the game.
367     *
368     * The loops do this by first going the gameboard through with for-loops and then utilising if-else to check,
369     * if the current mark is a mark placed by either of the palyers. If it is, the markCount increases with +1 until
370     * an empty mark is checked or the required amount of marks is achieved. markCount is reset when it comes across a '-'.
371     *
372     * @return either a mark of a player (that also determines the winner), or an empty mark if no winner is found
373     */
374    public static char CheckWins() {
375        // Keeps track of the amount of marks found in a horizontal, vertical or a diagonal line.
376        int markCount = 0;
377        // The last mark that was checked by the loop.
378        char markLast = markEmpty;
379        // Goes through the gameboard and checks for HORIZONTAL WINS.
380        for (int x = 0; x < gameboardHeight; x++) {
381            for (int y = 0; y < gameboardWidth; y++) {
382                // If the last mark checked was something else than a '-'.
383                if (Gameboard[x][y] != markEmpty) {
384                    // If the last mark checked also the same player's mark, markCount goes up by 1.
385                    if (Gameboard[x][y] == markLast) {
386                        markCount++;
387                    } else {
388                        // Mark count is kept at one, if there was just a singular player mark.
389                        markCount = 1;
390                        markLast  = Gameboard[x][y];
391                    }
392                    // If the amounts of subsequent marks equal the marks required to win,
393                    // changes the boolean value of hasWon to true (thus ending the game)
394                    // and sending the latest mark checked to main() and determining the winner.
395                    if (markCount == markRequired) {
396                        hasWon = true;
397                        return markLast;
398                    }
399                } else {
400                    // No wins found, resets the counter and the last mark checked to an empty one, ready for the next turn.
401                    markLast = markEmpty;
402                    markCount = 0;
403                }
404            }
405            // No wins found, resets the counter and the last mark checked to an empty one, ready for the next turn.
406            markLast = markEmpty;
407            markCount = 0;
408        }
409
410        // Does the same thing as the loop before this, but checks for VERTICAL WINS instead.
411        // x and y for-loops have swapped places, otherwise it's and identical loop with identical functions.
412        for (int y = 0; y < gameboardWidth; y++) {
413            for (int x = 0; x < gameboardHeight; x++) {
414                // If the last mark checked was something else than a '-'.
415                if (Gameboard[x][y] != markEmpty) {
416                    // If the last mark checked also the same player's mark, markCount goes up by 1.
417                    if (Gameboard[x][y] == markLast) {
418                        markCount++;
419                    } else {
420                        // Mark count is kept at one, if there was just a singular player mark.
421                        markCount = 1;
422                        markLast  = Gameboard[x][y];
423                    }
424                    // If the amounts of subsequent marks equal the marks required to win,
425                    // changes the boolean value of hasWon to true (thus ending the game)
426                    // and sending the latest mark checked to main() and determining the winner.
427                    if (markCount == markRequired) {
428                        hasWon = true;
429                        return markLast;
430                    }
431                } else {
432                    // No wins found, resets the counter and the last mark checked to an empty one, ready for the next turn.
433                    markLast = markEmpty;
434                    markCount = 0;
435                }
436            }
437            // No wins found, resets the counter and the last mark checked to an empty one, ready for the next turn.
438            markLast = markEmpty;
439            markCount = 0;
440        }
441        // Goes through the gameboard and checks for DIAGONAL WINS (from top-left to bottom-right / top-right to bottom-left).
442        for (int x = 0; x < gameboardHeight; x++) {
443            for (int y = 0; y < gameboardWidth; y++) {
444                // If the last mark checked was something else than a '-'.
445                if (Gameboard[x][y] != markEmpty) {
446                    // If the last mark checked also the same player's mark, markCount goes up by 1.
447                    if (Gameboard[x][y] == markLast) {
448                        markCount++;
449                    } else {
450                        // Mark count is kept at one, if there was just a singular player mark.
451                        // Diagonal checks are a bit trickier, so this is important...
452                        markCount = 1;
453                        markLast  = Gameboard[x][y];
454                        // This for-loop checks for top-left to bottom-right wins, by everytime adding the integer z to the width (y) & height (x)
455                        // (for as long as z is smaller than marks required to win, height+z smaller than max height, and width+z smaller than max width).
456                        for (int z = 1; z < markRequired && x+z < gameboardHeight && y+z < gameboardWidth; z++) {
457                            // If the mark found on the new coordinates equals to the last mark checked, markCount goes up by 1.
458                            if (Gameboard[x+z][y+z] == markLast) {
459                                markCount++;
460                                // If the amounts of subsequent marks equal the marks required to win,
461                                // changes the boolean value of hasWon to true (thus ending the game)
462                                // and sending the latest mark checked to main() and determining the winner.
463                                if (markCount == markRequired) {
464                                    hasWon = true;
465                                    return markLast;
466                                }
467                            } else {
468                                break;
469                            }
470                        }
471                        // Does the same thing as the previous loop that checked from top-left bottom-right,
472                        // this one checks from top-right to bottom-left by everytime adding the integer z to the height (x) and substracting z from the width (y).
473                        // (for as long as z is smaller than marks required to win, height+z smaller than max height, and width-z smaller than max width).
474                        for (int z = 1; z < markRequired && x+z < gameboardHeight && y-z < gameboardWidth; z++) {
475                            // An additional check, that the width is equal to or greater than the marks required to win (-1),
476                            // this to keep the loop from checking top-right to bottom-left victories while the mark checked is on coordinate,
477                            // from where a top-right to bottom-left victory is impossible to achieve (e.g. 5 marks to win, 'X' was checked on column 3).
478                            if (y >= markRequired-1) {
479                                // If the mark found on the new coordinates equals to the last mark checked, markCount goes up by 1.
480                                if (Gameboard[x+z][y-z] == markLast) {
481                                    markCount++;
482                                    // If the amounts of subsequent marks equal the marks required to win,
483                                    // changes the boolean value of hasWon to true (thus ending the game)
484                                    // and sending the latest mark checked to main() and determining the winner
485                                    if (markCount == markRequired) {
486                                        hasWon = true;
487                                        return markLast;
488                                    }
489                                } else {
490                                    break;
491                                }
492                            }
493                        }
494                        // No wins found, resets the counter and the last mark checked to an empty one, ready for the next turn.
495                        markLast = markEmpty;
496                        markCount = 0;
497                    }
498                }
499            }
500        }
501        // If no wins are found, returns an empty mark from the board.
502        return markEmpty;
503    }

505    /**
506     * Checks if the board is full or not.
507     * @return is the gameboard full and no more turns are possible, equaling in a draw.
508     */
509    public static boolean FullBoard() {
510        // By default, the value to return is "true".
511        isDraw = true;
512        // Goes through the board with for-loops, checking if empty marks are found.
513        for (int x = 0; x < gameboardHeight; x++) {
514            for (int y = 0; y < gameboardWidth; y++) {
515                if (Gameboard[x][y] == markEmpty) {
516                    // Empty marks found, changes the value of isDraw to "false".
517                    isDraw = false;
518                }
519            }
520        }
521        // Returns the value of isDraw to main().
522        return isDraw;
523    }

549    /**
550     * This loop checks the user input for the size of the gameboard.
551     * @param numToCheck either the value for the height or the width of the gameboard
552     * @return an integer equal to or bigger than 3, for either gameboardHeight or gameboardWidth
553     */
554    public static int numBoard(int numToCheck) {
555        Scanner input = new Scanner(System.in);
556        // While the number input by the user is under 3, a decimal number or not a number at all.
557        while (numToCheck < 3) {
558            // Checks if the value input by the user is an integer.

583     * @return an integer that is 3 or bigger and equal or smaller than the shorter side of the gameboard

593            shorter = numWidth;
594        } else {
595            shorter = numHeight;
596        }        
597        // While the number input by the user is under 3, bigger than the shorter side of the gameboard, a decimal number or not a number at all.
598        while (numToCheck < 3 || numToCheck > shorter) {

612                System.out.println();
613            }
